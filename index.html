function uncurry(f) {
  console.log("uncurry: ", f);
  return (...args) => {
    if (typeof f !== "function" )  return f;                        
    let fArgsLen = f.length;
    let fArgs = args.slice(0, fArgsLen);
    let f1 = f.apply(this, fArgs);
    let newArgs = args.slice(fArgsLen);
    return uncurry(f1)(...newArgs);
  };
